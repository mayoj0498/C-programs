Write a C program to calculate the factorial of a given number
Test Data :
Input the number : 5
Expected Output :
The Factorial of 5 is: 120
*/
#include<stdio.h>
int main(void)
{
    int num,i,fact=1;
    printf("please enter a number :");
    scanf("%d",&num);
    if(num>=0)
    {
        for(i=2;i<=num;i++)
        fact=fact*i;

        printf("\nFactorial of a given number : %d",fact);

    }
    else
    printf("\nInvalid input");
    return 0;
}
