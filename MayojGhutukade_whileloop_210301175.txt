To calculate and print the commission earned by a salesman based on criteria given
If sales <= Rs. 5000 commission rate =10% of sales
Otherwise commission rate = 20% of sales
*/
#include<stdio.h>
int main(void)
{
    float sale,comm;
    printf("please enter sales :");
    scanf("%f",&sale);
    while(sale>0)
    {
        if(sale<=5000)
        {
            comm=sale*0.1;
            printf("\ncommission earned : %.2f rs",comm);
        }
        else
        {
          comm=sale*0.2;
          printf("\ncommission earned : %.2f rs",comm);
        }  
        break;
        
    }
    return 0;
}

Write a program to print a number given by user but digits reversed. E.g.-
INPUT : 123 OUTPUT : 321
INPUT : 12345 OUTPUT : 54321
*/
#include<stdio.h>
int main(void)
{
    int num,rem,sum=0 ;
    printf("please enter a number :");
    scanf("%d",&num);
    while(num>0)
    {
        rem=num%10;
        sum=sum*10+rem;
        num=num/10;

    }
    printf("\n Reverse: %d",sum);
    return 0;
}
