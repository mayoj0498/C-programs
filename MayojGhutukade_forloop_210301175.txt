Print all special numbers between 1 to 200000
*/
#include<stdio.h>
int main(void)
{
    long i,j,sum,fact,rem,k;
    printf("all special numbers between 1 to 200000 :\n");
    for(i=2;i<200000;i++)
    {
        sum=0;
        
        for(j=i;j!=0;j=j/10)
        {
            fact=1;
            rem=j%10;
            for(k=rem;k>1;k--)
            fact=fact*k;
            sum=sum+fact;
        }
        if(i==sum)
        printf(" %ld ",i);

    }
    return 0;
}

*/
#include<stdio.h>
int main(void)
{
    int i,j,count;
    printf("Prime numbers between 51 to 100 : \n");
    for(i=51;i<=100;i++)
    {
        count=0;
        for(j=2;j<=i/2;j++)
        {
            if(i%j==0)
            {
                count=1;
                break;
            }
        }
        if(count==0)
        printf(" %d ",i);
    }

return 0;
}

Accept a number from the user and check if it is a prime number or not
*/
#include<stdio.h>
int main(void)
{
    int num,i,count=0;
    printf("please enter a number :");
    scanf("%d",&num);
    if(num>0)
    {
    for(i=2;i<=num/2;i++)
    {
        if(num%i==0)
        {
            count=1;
            break;
        }
    }
    if(num==1)
    printf("\n1 is neither prime nor composite");
    else if (count==0)
    printf("\n%d is a prime number",num);
    else
    printf("\n%d is not a prime number",num);
    }
    else
    printf("\nInvalid number");

return 0;
}

#include<stdio.h>
#include<math.h>
int main(void)
{
    long i,sum,power,j,k,rem;
    printf("Print all Armstrong number between 1 to 200000 : \n");
    for(i=2;i<200000;i++)
    {  
        sum=0;
        power=0;
        for(j=i;j!=0;j=j/10)
        { 
            power++;
        }
        for(k=i;k!=0;k=k/10)
        {
            rem=k%10;
            sum=sum+pow(rem,power);
        }
        if(sum==i)
        printf(" %ld ",i);
    }
return 0;
}
