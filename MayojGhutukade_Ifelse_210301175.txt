Accept the gross salary of an employee. Calculate and print the tax based on the given criteria :

Salary Tax rate
Up to Rs. 2,00,000 Nil
2,00,001 - 5,00,000 10%
Above 5,00,000 20%
*/
#include<stdio.h>
int main(void)
{
    double sal;
    printf("please enter gross salary : ");
    scanf("%lf",&sal);
    if(sal>=0) //validate salary
    {
        if(sal<=200000)
            printf("\nTax :0 rs");
        else if((sal>200000)&&(sal<=500000))
            printf("\nTax : %lf",0.1*sal);
        else 
            printf("\nTax : %lf",0.2*sal);
    }
    else
        printf("\nInvalid salary");
    return 0;
}

Accept the quantity of an item purchased and its price. Calculate the amount of purchase. If the amount

{
    int quant;
    float cp,amount,net,disc;
    printf("please enter quantity of an item purchased : ");
    scanf("%d",&quant);
    printf("please enter cost price of a single item : ");
    scanf("%f",&cp);
    if((quant>0)&&(cp>0)) //for validation of inputs
    {
    amount=quant*cp;
        if(amount>5000)
        {
        disc=20;
        printf("\nQuantity:%d   Amount:%f   Discount:%f",quant,amount,disc);
        net=amount-disc/100*amount;
        printf("\nNet amount : %f",net);
        }
        else
         {
        disc=10;
        printf("\nQuantity:%d   Amount:%f   Discount:%f",quant,amount,disc);
        net=amount-disc/100*amount;
        printf("\nNet amount : %f",net);
        }

    }
    else
    printf("\nInvalid input");
    return 0;
}

    scanf("%f",&s1);
    printf("\nSubject 2 : ");
    scanf("%f",&s2);
    printf("\nSubject 3 : ");
    scanf("%f",&s3);
    if(s1<=80&&s2<=80&&s3<=80) 
    {
        percent=(s1+s2+s3)/240*100;
        if(percent>=60)
        printf("\nVery Well Done");
        else
        printf("\nAverage performance");

    }
    else
    {
        printf("\nInvalid marks");
    }
    return 0;

}

Accept the age of a person and print an appropriate message as per the table given alongside:
Age Message
<=12 You are a child
13-19 You are a teenager
20-59 You are an adult
>=60 You are a senior citizen
*/
#include<stdio.h>
int main(void)
{
    int age;
    printf("please enter your age:");
    scanf("%d",&age);
    if(age>0) 
    {
        if(age>=60)
        printf("\nYou are a senior citizen");
        else if(age>=20&&age<=59)
        printf("\nYou are an adult");
        else if(age>=13&&age<=19)
        printf("\nYou are a teenager");
        else
        printf("\nYou are a child"); 
    }
    else
    printf("\nInvalid age");
    return 0;
}
