 Accept data in two 3*3 matrices and calculate the product of the matrices.   
*/

#include<stdio.h>
int main(void) 
{
   int a[5][5], b[5][5], c[5][5];
   int i, j, k,sum = 0;
   printf("\nplease enter First Matrix :\n");
   for (i = 0; i < 3; i++)
   {
      for (j = 0; j < 3; j++)
      scanf("%d", &a[i][j]);
   }
 
   printf("\nplease enter Second Matrix:\n");
   for (i = 0; i < 3; i++)
    {
      for (j = 0; j < 3; j++)
         scanf("%d", &b[i][j]);
   }

   for(i = 0; i <= 2; i++)
   {
      for (j = 0; j <= 2; j++) 
      {
       sum = 0;
         for (k = 0; k <= 2; k++)
         sum = sum + a[i][k] * b[k][j];
         c[i][j] = sum;
      }
   }
 
   printf("\nMultiplication Of Two Matrices :\n");
   for (i = 0; i < 3; i++) 
   {
      for (j = 0; j < 3; j++)
      printf(" %3d ", c[i][j]);
      printf("\n");
   }
 
   return 0;
}

Accept 10 numbers in an array; accept a number to be deleted from the array. Delete the number and print
the new array.
*/
#include<stdio.h>
int main(void)
{
    int arr[20],delete,pos=0,match=0,j,i,k;
    printf("\nEnter 10 numbers : ");
    for (i=0;i<10;i++)
        scanf("%d",&arr[i]);
    printf("\nEnter num to be deleted :");
    scanf("%d",&delete);
    for(j =0;j<10-match;j++)
    {
        if(delete == arr[j])
        {
            pos=j;
            match++;
            for (k=pos;k<10-match;k++)
                  arr[k]=arr[k+1];       
        }
    }
    if(pos==0)
        printf("\nNumber not found");

    printf("\nAfter Deletion : ");
    for (i=0;i<10-match;i++)
       printf("%d ",arr[i]);

    return 0;
}
